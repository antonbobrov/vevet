JavaScript
import {
  Preloader
} from "vevet";

const hideButton = document.getElementById("hide-preloader");

const preloader = new Preloader({
    container: document.getElementById("preloader"),
    hide: false
  },

  {
    onLoaded: () => {
      hideButton.classList.add("show");

      hideButton.addEventListener("click", () => {
        preloader.hide();
      });
    }
  });

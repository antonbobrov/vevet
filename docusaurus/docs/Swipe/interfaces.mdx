## ISwipeVec2

```ts
interface ISwipeVec2 {
  x: number;
  y: number;
}
```

## ISwipeVec3

```ts
interface ISwipeVec3 extends ISwipeVec2 {
  angle: number;
}
```

## ISwipeState

```ts
interface ISwipeState extends ISwipeVec3 {
  time: number;
}
```

## ISwipeBounds

```ts
interface ISwipeBounds {
  x?: number[];
  y?: number[];
  angle?: number[];
}
```

Each axis is a `[min, max]` pair in the same units as `diff` (px for `x`/`y`, degrees for `angle`).

## ISwipeCanMoveArg

```ts
interface ISwipeCanMoveArg {
  type: 'touch' | 'mouse';
  state: ISwipeState;
  start: ISwipeVec2;
  diff: ISwipeVec2;
}
```

## ISwipeCoords

```ts
interface ISwipeCoords {
  /** Start position. */
  start: ISwipeState;
  /** Previous position. */
  prev: ISwipeState;
  /** Current position. */
  current: ISwipeState;
  /** Movement offset from start. */
  diff: ISwipeState;
  /** Movement offset from previous position. */
  step: ISwipeState;
  /** Absolute path length since swipe start (`|step|` per axis). */
  accum: ISwipeVec3;
  /** Applied movement since instance init (after rubber). */
  total: ISwipeVec3;
}
```

StaticTimeline

StaticTimeline is a way to manipulate with progress.

Example

div
    input(
        type="range",
        id="timeline-range"
        min="0"
        max="1"
        step="0.0001"
        value="0.25"
        style="width: 100%;"
    )

div#timeline-visualizer.timeline-visualizer
    div#timeline-visualizer-thumb.timeline-visualizer-thumb

style 
    = readFile('./sample.css', dir)
.timeline-visualizer {
    position: relative;
    margin-top: 10px;
    width: 100%;
    height: 20px;
    background-color: #dedede;
}

.timeline-visualizer-thumb {
    position: absolute;
    top: -5px;
    left: 0;
    width: 10px;
    height: 30px;
    transform: translate(-50%, 0);
    background-color: #000;
}

import { Easing } from 'easing-progress';
import * as Vevet from '../../../../../src/ts/index';

new Vevet.Application();

// get visualizer
const visualizerThumb = document.querySelector('#timeline-visualizer-thumb');

// create a timeline
const timeline = new Vevet.StaticTimeline({
    easing: Easing.easeInOutBounce,
});
timeline.addCallback('progress', (data) => {
    if (visualizerThumb) {
        visualizerThumb.style.left = `${data.easing * 100}%`;
    }
});

const rangeInput = document.querySelector('#timeline-range');
if (rangeInput) {
    timeline.progress = parseFloat(rangeInput.value);
    rangeInput.addEventListener('input', () => {
        timeline.progress = parseFloat(rangeInput.value);
    });
}
